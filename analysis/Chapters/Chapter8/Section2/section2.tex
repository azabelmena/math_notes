\section{$\sigma$-algebras of Lebesuge Measurable Sets}

\begin{definition}
    We say a set $E$ is  \textbf{measurable} if for any $A \subseteq \R$, then:
    \begin{equation*}
        m^*(A)=m^*(A \cap E)+m*(A \cap (\com{A}{E}))
    \end{equation*}
\end{definition}

\begin{lemma}\label{}
    Any set of outer measure $0$ is measurable.
\end{lemma}

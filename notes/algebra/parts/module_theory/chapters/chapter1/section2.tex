\section{Algebras over Commutative Rings}

\begin{definition}
  Let $R$ be a commutative ring with identity. An
  \textit{$R$-algebra} (or \textit{algebra over $R$}) is a ring $A$
  with identity together with a ring homomorphism  $f:R
  \xrightarrow{} A$ (taking $1_R \xrightarrow{} 1_A$) such that the
  subring $f(R)$ is contained in the center of $A$.
\end{definition}

\begin{lemma}\label{lemma_9.2.1}
  Let $R$ be a commutative ring with identity. If $A$ is an
  $R$-algebra, then $A$ is a unital $R$-module under the action:
  \begin{align*}
    R \times A  & \xrightarrow{} A  \\
    (r,a)   & \xrightarrow{} ra=f(r)a \\
  \end{align*}
\end{lemma}

\begin{definition}
  Let $R$ be a commutative ring with identity. If $A$ and $B$ are
  $R$-algebras, and \textit{$R$-algebra homomorphism} is a ring
  homomorphism $\phi:A \xrightarrow{} B$ such that for every $r \in
  R$ and $a \in A$:
  \begin{enumerate}
    \item[(1)] $\phi(1_A)=1_B$.

    \item[(2)] $\phi(ra)=r\phi(a)$.
  \end{enumerate}
  We call $\phi$ an  \textit{$R$-algebra isomorphism} provided
  $\phi$ is a ring isomorphism, and we call $A$ and $B$
  \textit{isomorphic} and write $A \simeq B$.
\end{definition}

\begin{example}\label{example_9.5}
  Let $R$ be any commutative ring with identity, and let $f$ denote
  any ring homomorphism from $R$ to a given ring $A$  (not
  necessarily an $R$-algebra).
  \begin{enumerate}
    \item[(1)] Any ring $A$ with identity is a $\Z$-algebra.

    \item[(2)] For any ring $A$ with identity, if $f(R) \subseteq
      Z(A)$ is a subring containing $1_A$, then $A$ can be made into
      an $R$-algebra.

    \item[(3)] The polynomial ring $R[x]$ is an $R$-algebra as is
      the multivariate polynomial ring $R[x_1, \dots, x_n]$.

    \item[(4)] For a finite group $G$, the group ring $RG$ is an
      $R$-algebra.

    \item[(5)] For any $R$-algebra $A$, its $R$-module structure
      depends only on the subring $f(R)$. Hence, up to ring
      homomorphism, every $R$-algebra $A$ arises from a subring
      contained in its center containing $1_A$.
  \end{enumerate}
\end{example}

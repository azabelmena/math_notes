\section{Coordinate Rings and Polynomial Maps}\label{section_2.1}

\begin{definition}
  We call an irreducible affine algebraic set in $\A^n(k)$ an
  \textit{affine variety}.
\end{definition}

\begin{definition}
  We define the \textit{coordinate ring} of an affine variety $V
  \subseteq \A^n(k)$ to be the following quotient ring:
  \begin{equation*}
    k[V]=\faktor{k[x_1, \dots, x_n]}{I(V)}
  \end{equation*}
\end{definition}

\begin{definition}
  Let $V \subseteq \A^n(k)$ be an affine variety. We call a function
  $F:V \xrightarrow{} k$ a \textbf{polynomial function} if there
  exists a polynomial $f(x_1, \dots, x_n) \in k[x_1, \dots, x_n]$
  for which $F(P)=f(P)$ for all $P \in V$.
\end{definition}

\begin{proposition}\label{proposition_11.1.1}
  The set of all polynomial functions on an affine variety $V
  \subseteq \A^n(k)$ forms a ring containing $k$.
\end{proposition}
\begin{proof}
  Let $F,G:V \xrightarrow{} k$ polynomial functions. Then there
  exists $f(x_1, \dots, x_n),g(x_1, \dots, x_n) \in k[x_1, \dots,
  x_n]$ for which $F(P)=f(P)$ and $G(P)=g(P)$ for all $P \in V$.
  Then $(F-G)(P)=F(P)-G(P)=f(P)-g(P)=(f-g)(P)$ for all $P \in V$.
  Since $f-g \in k[x_1, \dots, x_n]$, $F-G:V \xrightarrow{} k$ is a
  polynomial function. By similar reasoning, we see that the
  function $FG:V \xrightarrow{} k$ is also a polynomial function.

  Now, take the function $c:V \xrightarrow{} k$ by $F(P)=a_1$ for
  any point $P=(a_1, \dots, a_n)$. Then $c$ is a polynomial function
  determined by the constant polynomial $c_{a_1}(x_1, \dots, x_n) \in
  k[x_1, \dots, x_n]$. Thus, there exists an imbedding taking $k$
  into the ring of polynomial functions on $V$.
\end{proof}
\begin{corollary}
  Any two polynomials $f(x_1, \dots, x_n)$ and $g(x_1, \dots, x_n)$
  determine the same polynomial function on $V$ if, and only if
  $f(x_1, \dots, x_n) \equiv g(x_1, \dots, x_n) \mod{I(V)}$.
\end{corollary}
\begin{proof}
  Indeed, let $F:V \xrightarrow{} k$ be such that $f(P)=F(P)$ and
  $g(P)=G(P)$ for all $P \in V$. Then
  $(f-g)(P)=f(P)-g(P)=F(P)-F(P)=0$, so that $(f-g)(P)=0$ for all $P
  \in V$. This puts $f-g \in I(V)$, so that $f(x_1, \dots, x_n) \equiv
  g(x_1, \dots, x_n) \mod{I(V)}$. Conversely, if $f(x_1, \dots, x_n)
  \equiv g(x_1, \dots, x_n) \mod{I(V)}$, then for any $P \in V$,
  $f(P)=g(P)$. If $f(x_1, \dots, x_n)$ determines the polynomial
  function $F$, then so does $g(x_1, \dots, x_n)$.
\end{proof}
\begin{corollary}
  Let $f(x_1, \dots, x_n) \in k[x_1, \dots, x_n]$, and consider the
  map $f \xrightarrow{} F$ sending $f(x_1, \dots, x_n)$ to its associated
  polynomial function $F$. This map is a ring homomorphism, and has
  as its kernel $I(V)$.
\end{corollary}
\begin{proof}
  Define the map $\phi:f \xrightarrow{} F$ where $f(x_1, \dots, x_n)
  \in k[x_1, \dots, x_n]$, and $F$ is the polynomial function
  determined by $f(x_1, \dots, x_n)$. Pick a $g(x_1, \dots ,x_n) \in
  k[x_1, \dots, x_n]$, and $G$ its associated polynomial function.
  And let $F+G$ the polynomial function associated to the polynomial
  $f+g$. Fix $P  \in V$, then $\phi(f+g)(P)= F G)(P)=F(P)+G(P)=
  \phi(f)(P)+\ph (g)(P)$. Similarly, letting $FG$ be the polynomial
  function determined by the polynomial $fg$, we see that
  $\phi(fg)=\phi(f)\phi(g)$. Hence, $\phi$ is a homomorphism.

  Now, let $f(x_1, \dots, x_n) \in \ker{\phi}$. Then $\phi(f)=0$, so by
  the above corollary, $f(x_1, \dots, x_n) \equiv 0 \mod{I(V)}$, and
  hence $f(x_1, \dots, x_n) \in I(V)$ so that $\ker{\phi}=I(V)$.
\end{proof}

From the above proposition, and its corollaries, we can talk about
polynomial functions on an affine variety $V$ in terms of the
polynomials in $k[V]$. We may also speak about ``subvarieties'' of
$V$ by observing the coordinate ring $k[V]$.

\begin{definition}
  Let $V \subseteq \A^n(k)$ an affine variety. An \textit{affine
  subvariety} of $V$ is an affine variety which is contained in $V$.
\end{definition}

\begin{proposition}\label{proposition_11.1.2}
  There exists a natural $1-1$ correspondence between algebraic
  subsets of an affine variety  $V$, and radical ideals of $k[V]$.
\end{proposition}
\begin{proof}
  This follows to Hilbert's Nullstellensatz restricted to the
  coordinate ring $k[V]$.
\end{proof}
\begin{corollary}
  There exists a $1-1$ correspondence between subvarieties of $V$
  and prime ideals in  $k[V]$.
\end{corollary}
\begin{corollary}
  There exists a $1-1$ correspondence between points of $V$ and
  maximal ideals in $k[V]$.
\end{corollary}

\begin{proposition}\label{proposition_11.1.3}
  Let $V \subseteq \A^n(k)$ be an affine variety and let $W
  \subseteq V$ an affine subvariety. Let $I_V(W)$ denote the ideal
  of $W$ in $k[V]$. Then any polynomial function on $V$ restricts to
  a polynomial function on $W$. Moreover, the map $k[V]
  \xrightarrow{} k[W]$ defined by these restrictions defines a
  homomorphism whose kernel is $I_V(W)$.
\end{proposition}
\begin{proof}
  Let $f(x_1, \dots, x_n) \in k[x_1, \dots, x_n]$, and let $F:V
  \xrightarrow{} k$ be the polynomial function determined by $f(x_1,
  \dots, x_n)$. Then $f(P)=F(P)$ for all $P \in V$. Define the
  polynomial $g(x_1, \dots, x_n) \in k[W]$, such that $f(P)=g(P)$
  for all $P \in W$. Associate to $g(x_1, \dots, x_n)$ the
  polynomial map $G:W \xrightarrow{} k$. Then $G(P)=g(P)=f(P)=F(P)$
  for all $P \in W$. That is $G(P)=F(P)$ for all $P \in W$,
  therefore $G$ is a restriction of $F$.

  Now, define the map $\phi:k[V] \xrightarrow{} k[W]$ by $\phi:g
  \xrightarrow{} g$, where $g(P)=f(P)$ for all $P \in W$. $\phi$ is
  a homomorphism of polynomials. Now, suppose that $f(x_1, \dots,
  x_n) \in \ker{\phi}$, so that $\phi(f)=0$, so taht $f(P)=0$ for
  all $P \in W$. This puts $f(x_1, \dots, x_n) \equiv 0
  \mod{I_V(W)}$, so that $f(x_1, \dots, x_n) \in I_V(W)$. Therefore
  $\ker{\phi}=I_V(W)$.
\end{proof}
\begin{corollary}
  $k[W] \simeq \faktor{k[V]}{I_V(W)}$.
\end{corollary}

\begin{proposition}\label{proposition_11.1.4}
  Let $V \subseteq \A^n(k)$ be an affine variety. The following
  statements are equivalent:
  \begin{enumerate}
    \item[(1)] $V$ is a point in $\A^n(k)$.

    \item[(2)] $k[V] \simeq k$.

    \item[(3)] $\dim_k{k[V]}$ is finite.
  \end{enumerate}
\end{proposition}
\begin{proof}
  Suppose that $V=\{P\}$ for some point $P \in \A^n(k)$. Then $I(V)$
  is a maximal ideal, which makes $k[V]$ into a field. It follows
  then that $k[V] \simeq k$. The converse of this argument also
  follows.

  Now, we also have that if $V=\{P\}$, then $V$ is a finite set of
  points, and  $\dim_k{k[V]}$ is finite by proposition
  \ref{proposition_10.5.3}. Now, if $k[V] \simeq k$, then
  $\dim_k{k[V]}=\dim_k{k}=1$. This completes the proof.
\end{proof}

\begin{example}\label{example_11.1}
  Let $f(x,y) \in k[x,y]$ an irreducible polynomial monic in $y$;
  that is
  \begin{equation*}
    f(x,y)=a_0(x)+a_1(x)+\dots+a_{n-1}(x)y^{n-1}+y^n \in (k[x])[y]
  \end{equation*}
  Let $V=V(f)$ in $\A^2(k)$, and let $\pi:k[x,y] \xrightarrow{}
  k[V]$ be the natural homomorphism. Let $g(x,y) \in \ker{\pi}$, so
  that $g(x,y) \equiv 0 \mod{f}$. Then $g(x,y) \in (f)$, hence
  $\ker{\pi}=(f)=I(V)$, this makes $\pi$  $1-1$. Moreover, since
  $f(x,y)$ is monic in $y$, $k[x]$ can be imbedded into $k[V]$ via
  $\pi$ as a coefficient ring.

  Now, let $Y=\{1, y, \dots, y^{n-1}\}$. Since $f(x,y)$ is monic in
  $y$, then for every  $g(x,y) \in k[V]$, $g(x,y)$ has the form:
  \begin{equation*}
    g(x,y)=b_0(x)+b_1(x)y+\dots+b_{n-1}(x)y^{n-1}
  \end{equation*}
  and where $y^n=-a_0(x)-a_1(x)y-\dots-a_{n-1}y^{n-1}$. Hence all
  elements of $k[V]$ over $k[x]$ are of the form
  $\sum_{i=0}^{n-1}{a_i(x)y^i}$, where $a_i(x) \in k[x]$ and $y^i
  \in Y$ for all $0 \leq i \leq n-1$. This makes $Y$ a generating
  set of $k[V]$ as a module over $k[x]$.
\end{example}
